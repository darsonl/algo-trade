    ```python
    def get_top_sp500_by_fundamentals(n: int = 10) -> list[str]:
        """
        Return top N S&P 500 tickers ranked by combined EPS + ROE score.
        Results are cached in-memory for 24 hours.
        """
        global _top_sp500_cache, _top_sp500_cache_time
        import logging
        logger = logging.getLogger(__name__)

        if _top_sp500_cache is not None and (_time.time() - _top_sp500_cache_time) < _CACHE_TTL_HOURS * 3600:
            return _top_sp500_cache[:n]

        sp500 = get_sp500_tickers()
        if not sp500:
            logger.warning("No S&P 500 tickers available for ranking")
            return []

        import yfinance as yf
        scores: list[tuple[str, float]] = []
        for ticker in sp500:
            try:
                info = yf.Ticker(ticker).info
                eps = info.get("trailingEps")
                roe = info.get("returnOnEquity")
                if eps is None or roe is None:
                    continue
                scores.append((ticker, eps, roe))
            except Exception as exc:
                logger.debug("Skipping %s in ranking: %s", ticker, exc)
                continue

        if not scores:
            logger.warning("No tickers had valid EPS+ROE data")
            return []

        # Rank by EPS (descending) and ROE (descending), then combine ranks
        sorted_by_eps = sorted(scores, key=lambda x: x[1], reverse=True)
        sorted_by_roe = sorted(scores, key=lambda x: x[2], reverse=True)

        eps_rank = {t[0]: i for i, t in enumerate(sorted_by_eps)}
        roe_rank = {t[0]: i for i, t in enumerate(sorted_by_roe)}

        combined = sorted(eps_rank.keys(), key=lambda t: eps_rank[t] + roe_rank[t])

        _top_sp500_cache = combined
        _top_sp500_cache_time = _time.time()

        return combined[:n]
    ```

    Key behaviors:
    - Skips tickers where yfinance returns `None` for `trailingEps` or `returnOnEquity` (logs at debug level)
    - Combined rank = `eps_rank + roe_rank` where lower = better (rank 0 = best)
    - On empty S&P 500 list or no valid scores, returns empty list with warning log
    - Cache is module-level `_top_sp500_cache` / `_top_sp500_cache_time` using `time.time()` for TTL check
    - Returns `combined[:n]` (sliced to requested count)
