"""Tests for screener/exit_signals.py — SELL-02.

Gate logic: RSI > threshold AND macd_line < signal_line (both required per D-01/D-10).
"""
from config import Config
from screener.exit_signals import check_exit_signals

# Helpers for readability
OVERBOUGHT = {"rsi": 75.0, "macd_line": -0.5, "signal_line": 0.2}   # RSI high, MACD bearish
RSI_HIGH_MACD_BULL = {"rsi": 75.0, "macd_line": 0.5, "signal_line": 0.2}  # RSI high, MACD bullish
RSI_LOW_MACD_BEAR = {"rsi": 50.0, "macd_line": -0.5, "signal_line": 0.2}  # RSI low, MACD bearish
RSI_LOW_MACD_BULL = {"rsi": 50.0, "macd_line": 0.5, "signal_line": 0.2}   # both calm


def test_rsi_high_macd_bearish_returns_true():
    """Both gates pass — should trigger sell."""
    config = Config()
    assert check_exit_signals(OVERBOUGHT, config) is True


def test_rsi_high_macd_bullish_returns_false():
    """RSI overbought but MACD still bullish — uptrend may continue, no signal."""
    config = Config()
    assert check_exit_signals(RSI_HIGH_MACD_BULL, config) is False


def test_rsi_low_macd_bearish_returns_false():
    """MACD bearish but RSI not overbought — no signal."""
    config = Config()
    assert check_exit_signals(RSI_LOW_MACD_BEAR, config) is False


def test_rsi_low_macd_bullish_returns_false():
    """Neither gate passes."""
    config = Config()
    assert check_exit_signals(RSI_LOW_MACD_BULL, config) is False


def test_rsi_at_threshold_returns_false():
    """RSI exactly at threshold is not strictly greater — no signal."""
    config = Config()
    config.sell_rsi_threshold = 70.0
    assert check_exit_signals({"rsi": 70.0, "macd_line": -0.5, "signal_line": 0.2}, config) is False


def test_rsi_none_returns_false():
    config = Config()
    assert check_exit_signals({"rsi": None, "macd_line": -0.5, "signal_line": 0.2}, config) is False


def test_macd_line_none_returns_false():
    config = Config()
    assert check_exit_signals({"rsi": 75.0, "macd_line": None, "signal_line": 0.2}, config) is False


def test_signal_line_none_returns_false():
    config = Config()
    assert check_exit_signals({"rsi": 75.0, "macd_line": -0.5, "signal_line": None}, config) is False


def test_empty_dict_returns_false():
    config = Config()
    assert check_exit_signals({}, config) is False


def test_custom_threshold():
    config = Config()
    config.sell_rsi_threshold = 80.0
    # RSI 75 < 80 → no trigger even with bearish MACD
    assert check_exit_signals({"rsi": 75.0, "macd_line": -0.5, "signal_line": 0.2}, config) is False
    # RSI 85 > 80 + bearish MACD → trigger
    assert check_exit_signals({"rsi": 85.0, "macd_line": -0.5, "signal_line": 0.2}, config) is True
